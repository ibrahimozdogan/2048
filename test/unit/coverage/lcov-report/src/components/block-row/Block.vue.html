<template>
    <div class="block" :style="{ backgroundColor }">
        {{value || ''}}
    </div>
</template>
 
<script>
    import CommonEnums from '../../enums/CommonEnums';
 
    export default {
        props: {
            /**
             * @property {number} value
             */
            value: {
                type: Number,
                required: true
            }
        },
 
        computed: {
            backgroundColor () {
                const color = CommonEnums.COLORS[this.value];
 
                return this.value === 0 ? '#CFC1B2' : (color || '#3E3933');
            }
        }
    }
</script>
